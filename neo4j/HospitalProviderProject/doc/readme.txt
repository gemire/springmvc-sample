        </map>
    </constructor-arg>     
</bean>

</bean>

function recurseName(item,searchString)
{
	 if (item == null)
	    return null;
	    
	 if (item.name == searchString)
	 {
		 return item;
	 }
	 for (i=0;i<item.children.length;i++)
	 {
		return recurseName(item.children[i],searchString);
		
	 }
